コールバック方式の原理

void blink_led() {
    turn_on(led1);
    sleep(500);
    turn_off(led1);
    sleep(500);
}
  • 下記のようにコードを変換する
  • sleep の後で第2引数で指定された関数をコールバックする
void blink_led_1() {
    turn_on(led1);
    sleep(500, blink_led_2);
}

void blink_led_2() {
    turn_off(led2);
    sleep(500, blink_led_3);
}

void blink_led_3() {}
