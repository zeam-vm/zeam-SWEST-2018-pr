  __global long* input,
  __global long* output) {
  size_t i = get_global_id(0);
  long temp = input[i];
  temp = foo(temp);        # 実際には関数呼出しをインライン展開することで，
  temp = bar(temp);        # さらなる最適化を促進する
  output[i] = temp;
}
